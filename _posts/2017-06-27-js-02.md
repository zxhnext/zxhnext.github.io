## 6.字符串截取方法
```js
getCharactersLen: function (charStr, cutCount) {
   if (charStr == null || charStr == '') return '';
   var totalCount = 0;
   var newStr = '';
   for (var i = 0; i < charStr.length; i++) {
       var c = charStr.charCodeAt(i);
       if (c < 255 && c > 0) {
           totalCount++;
       } else {
           totalCount += 2;
       }
       if (totalCount >= cutCount) {
           newStr += charStr.charAt(i);
           break;
       }
       else {
           newStr += charStr.charAt(i);
       }
   }
   return newStr;}　　
```
